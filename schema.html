  current_value        NUMERIC(10,4),

  breach_date          DATE,
  cure_period_days     INTEGER,
  waiver_granted       BOOLEAN        DEFAULT false,
  waiver_expiry        DATE,
  waiver_notes         TEXT,
  test_frequency       TEXT           CHECK(test_frequency IN ('MONTHLY','QUARTERLY','ANNUAL')),
  last_tested          DATE,
  created_at           TIMESTAMPTZ    DEFAULT NOW(),
  updated_at           TIMESTAMPTZ    DEFAULT NOW()
);
-- Auto-update is_breached via trigger
CREATE OR REPLACE FUNCTION check_covenant_breach() RETURNS TRIGGER AS $$
BEGIN
  IF NEW.threshold_direction = 'MIN' AND NEW.current_value < NEW.threshold_value THEN
    NEW.is_breached := true;
    IF OLD.is_breached = false THEN NEW.breach_date := CURRENT_DATE; END IF;
  ELSIF NEW.threshold_direction = 'MAX' AND NEW.current_value > NEW.threshold_value THEN
    NEW.is_breached := true;
    IF OLD.is_breached = false THEN NEW.breach_date := CURRENT_DATE; END IF;
  ELSE
    NEW.is_breached := false;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trig_covenant_breach
  BEFORE INSERT OR UPDATE ON covenants
  FOR EACH ROW EXECUTE FUNCTION check_covenant_breach();

-- ===== USEFUL VIEWS =====
CREATE VIEW v_portfolio_summary AS
  SELECT
    p.alert_level,
    COUNT(*) AS project_count,
    SUM(l.outstanding_amount) AS total_exposure,
    AVG(p.risk_score) AS avg_risk_score,
    AVG(p.dscr) AS avg_dscr,
    AVG(p.ltv) AS avg_ltv
  FROM projects p
  LEFT JOIN loans l ON p.loan_id = l.id
  GROUP BY p.alert_level;

CREATE VIEW v_active_breaches AS
  SELECT c.*, l.project_id, p.name AS project_name, p.alert_level
  FROM covenants c
  JOIN loans l ON c.loan_id = l.id
  JOIN projects p ON l.project_id = p.id
  WHERE c.is_breached = true AND c.waiver_granted = false;

CREATE VIEW v_upcoming_maturities AS
  SELECT l.*, p.name, p.alert_level, d.name AS developer_name
  FROM loans l
  JOIN projects p ON l.project_id = p.id
  JOIN developers d ON l.developer_id = d.id
  WHERE l.end_date <= CURRENT_DATE + INTERVAL '90 days'
    AND l.status NOT IN ('CLOSED','NPA')
  ORDER BY l.end_date ASC;
